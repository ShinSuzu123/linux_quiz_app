もし、解答や解説の表示に関してサーバーへのデータ送信が必要ない場合、つまりサーバーサイドの処理が不要であれば、JavaScriptだけで対応できます。以下に、各問題に対して解答と解説を表示する方法を示します。

$(document).ready(function() {
    // カテゴリーを指定して問題を取得
    var category = "コマンド基礎とファイル操作";

    $.get("get_questions.php?category=" + category, function(data) {
        if (data && data.length > 0) {
            for (var i = 0; i < data.length; i++) {
                var question = data[i];

                var questionHTML = '<div class="question">';
                // 問題
                questionHTML += '<h4>' + (i + 1) + '.' + question.question_text + '</h4>';
                // 選択肢
                questionHTML += ' <input type="radio" name="ans" value="1"> ' + question.option1;
                questionHTML += ' <input type="radio" name="ans" value="2"> ' + question.option2;
                questionHTML += ' <input type="radio" name="ans" value="3"> ' + question.option3;
                questionHTML += ' <input type="radio" name="ans" value="4"> ' + question.option4;

                // 解答と解説表示用のコンテナ
                questionHTML += '<div class="answer-container" id="answer-container-' + question.id + '">';
                questionHTML += '<button class="show-answer">解答と解説を表示</button>';
                questionHTML += '<div class="answer-content" style="display: none;">';
                questionHTML += '<p>解答: ' + question.correct_option + '</p>';
                questionHTML += '<p>解説: ' + question.explanation + '</p>';
                questionHTML += '</div></div>';

                questionHTML += '</div>';
                // questions-containerに追加
                $("#questions-container").append(questionHTML);
            }

            // 解答と解説表示ボタンのクリックイベント
            $('.show-answer').click(function() {
                var answerContainer = $(this).siblings('.answer-content');
                answerContainer.slideToggle();
            });
        } else {
            console.log("問題データが取得できません");
        }
    });
});
このコードでは、各問題ごとに「解答と解説を表示」ボタンを用意し、その下に解答と解説を表示する領域を用意しています。ボタンをクリックすることで、対応する解答と解説が表示・非表示されるようになります。データはすでに問題データから取得しているため、サーバーへの追加のデータ送信は不要です。
